- 40道选择题+两道编程题

- 编程题一为验证密码的复杂性

```c++
#include <iostream>
#include <string>
#include <set>
using namespace std;

string special = "([~!@#$%^&*()_+-*/<>,.[]\/])+";
set<char> st;

bool judge(string pwd) {
    if (pwd.size() < 8) {
        return false;
    }
    bool hasupper = false;
    bool haslower = false;
    bool hasdigit = false;
    bool hasspecial = false;
    for (char ch : pwd) {
        if (ch >= '0' && ch <= '9') {
            hasdigit = true;
        } else if (ch >= 'a' && ch <= 'z') {
            haslower = true;
        } else if (ch >= 'A' && ch <= 'Z') {
            hasupper = true;
        } else if (st.count(ch)) {
            hasspecial = true;
        }
    }
    return hasupper && haslower && hasdigit && hasspecial; 
}

int main() {
    string pwd;
    for (char ch : special) {
        st.insert(ch);
    }
    while (cin >> pwd) {
        if (judge(pwd)) {
            cout << "OK" << endl;
        } else {
            cout << "Irregular password" << endl;
        }
    }
    return 0;
}
```
